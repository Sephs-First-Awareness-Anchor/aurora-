# Aurora core dependencies
# ------------------------
# The minimal set required to import and boot the L0-L8 cognitive stack
# (`import aurora`, boot_aurora, process_external_user_turn).
#
# Everything in the flat `current_requirements.txt` beyond this is for an
# optional feature group (sensory/voice, GUI hub/room, training/evolution,
# cloud/API, mobile build) and is imported lazily at the point of use. CI and
# headless deployments only need this file.
#
# Verified: `AURORA_SKIP_DEP_INSTALL=1 python -c "import aurora"` succeeds with
# only numpy present.
#
# opencv-python-headless: needed by aurora_concept_imager.py's
# ingest_concept_image() -- the real (non-hardware) sensory grounding route
# the scheduled CI run uses in place of a live camera/mic (see
# scripts/aurora_ci_segment.py). Headless build avoids the GUI/libGL
# dependencies opencv-python pulls in, which a minimal CI runner doesn't have.
#
# Pinned to 4.11.0.86, NOT the latest (4.12.0.88+): those require
# numpy>=2, which conflicts with numpy==1.26.4 above and makes this file
# entirely uninstallable (`pip install -r requirements-core.txt` fails
# with ResolutionImpossible before Aurora ever boots). Confirmed via
# PyPI metadata: 4.11.0.86 requires numpy>=1.23.5 for python>=3.11, no
# upper bound -- the last version compatible with the 1.26.4 pin. If
# numpy ever gets upgraded to 2.x here, opencv can move too, together.

numpy==1.26.4
opencv-python-headless==4.11.0.86

# requests: needed by aurora_concept_imager.py's fetch_concept_image() --
# the Wikipedia REST API call that fetches the source image before
# ingest_concept_image() ever runs. Missing from this file since the module
# was added, so fetch_concept_image() silently returned None every run
# ("[IMAGER] requests not available") and every SEMANTIC+ candidate got
# marked permanently "failed" -- concept_images_ingested read 0 on every
# scheduled run regardless of any other fix to the ingestion path itself.
requests==2.32.3
